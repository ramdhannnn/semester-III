// Array data siswa const nama = ["Asep", "Iwan", "Cepi", "Agus", "Dadang"]; const tugas = [80, 75, 70, 65, 60]; const uts = [95, 75, 80, 40, 70]; const uas = [85, 90, 90, 55, 60]; // Array untuk menyimpan nilai akhir (NA) const nilaiAkhir = []; // Hitung nilai akhir untuk setiap siswa // Rumus: na = 40% tugas + 30% uts + 30% uas for (let i = 0; i < nama.length; i++) { const na = (0.4 * tugas[i]) + (0.3 * uts[i]) + (0.3 * uas[i]); nilaiAkhir.push(na); } // Menampilkan hasil dalam tabel HTML // Anda bisa menyalin dan menempelkan seluruh kode ini ke dalam file HTML // dan membukanya di browser untuk melihat hasilnya. // Mulai membuat output HTML let outputHTML = `

TABEL NILAI SISWA

`; // Lakukan perulangan untuk membuat baris tabel for (let i = 0; i < nama.length; i++) { outputHTML += ` `; } // Tutup tabel dan body HTML outputHTML += `
NO NAMA SISWA TUGAS UTS UAS NA
${i + 1} ${nama[i]} ${tugas[i]} ${uts[i]} ${uas[i]} ${nilaiAkhir[i].toFixed(1)}
`; // Tampilkan HTML yang dihasilkan di browser document.body.innerHTML = outputHTML;
